# ---------- 4) Escribir de vuelta al GeoJSON (versión robusta con agregación) ----------
# 1) Agrega por ADM2_PCODE para resolver municipios con varios polígonos
def first_non_null_as_str(s):
    s = s.dropna()
    if len(s) == 0:
        return None
    # si viene como Timestamp -> str(YYYY-MM-DD)
    v = s.iloc[-1]
    try:
        import pandas as pd
        if isinstance(v, pd.Timestamp):
            return str(v.date())
    except Exception:
        pass
    v = str(v)
    return None if v in ("nan", "None", "NaT") else v

agg = g.groupby("ADM2_PCODE", as_index=False).agg({
    "Fecha_pred": first_non_null_as_str,  # última no nula como string (o None)
    "Pred_mm": "median",                  # misma pred para todo el municipio → mediana/mean vale
    "Media_hist_mm": "median",
    "Diff_pct": "median"
})

# 2) Diccionarios por código para escritura rápida
d_fecha = dict(zip(agg["ADM2_PCODE"], agg["Fecha_pred"]))
d_pred  = dict(zip(agg["ADM2_PCODE"], agg["Pred_mm"]))
d_hist  = dict(zip(agg["ADM2_PCODE"], agg["Media_hist_mm"]))
d_diff  = dict(zip(agg["ADM2_PCODE"], agg["Diff_pct"]))

# 3) Vuelca a las properties de cada Feature
for f in fc["features"]:
    p = f["properties"]
    code = str(p.get("ADM2_PCODE", "")).strip().upper()
    # usa dicts; si no existe -> None
    fecha = d_fecha.get(code)
    pred  = d_pred.get(code)
    hist  = d_hist.get(code)
    diff  = d_diff.get(code)

    p["Fecha_pred"]    = None if fecha in (None, "nan", "None", "NaT") else str(fecha)
    p["Pred_mm"]       = None if pd.isna(pred) else float(pred)
    p["Media_hist_mm"] = None if pd.isna(hist) else float(hist)
    p["Diff_pct"]      = None if pd.isna(diff) else float(diff)

from pathlib import Path
import json, pandas as pd, numpy as np, re

GEO = Path("data/external/ven_adm2_with_preds.geojson")
RAW = Path("data/raw/ven-rainfall-adm2-full.csv")  # para 'Media_hist_mm' si existe

def gid2_state_id(gid2: str):
    if not isinstance(gid2, str): return None
    m = re.match(r"^VEN\.(\d+)\.", gid2.upper().strip())
    return int(m.group(1)) if m else None

# 1) Cargar GeoJSON existente
fc = json.load(open(GEO, "r", encoding="utf-8"))
props = [f["properties"] for f in fc["features"]]
df = pd.DataFrame(props)

# Normalizar tipos
df["ADM2_PCODE"] = df["ADM2_PCODE"].astype(str).str.strip().str.upper()
df["Pred_mm"] = pd.to_numeric(df["Pred_mm"], errors="coerce")
df["Media_hist_mm"] = pd.to_numeric(df.get("Media_hist_mm", np.nan), errors="coerce")
df["STATE_ID"] = df["GID_2"].map(gid2_state_id)

# 2) Si existe CSV histórico, calcula Media_hist_mm para completar donde falte
if RAW.exists():
    try:
        h = pd.read_csv(RAW, usecols=["ADM2_PCODE","r1q"], low_memory=False)
        h["ADM2_PCODE"] = h["ADM2_PCODE"].astype(str).str.strip().str.upper()
        hist_mean = h.groupby("ADM2_PCODE", as_index=False)["r1q"].mean().rename(columns={"r1q":"Media_hist_mm"})
        df = df.merge(hist_mean, on="ADM2_PCODE", how="left", suffixes=("", "_from_hist"))
        # si no había Media_hist_mm en props, úsala desde hist
        df["Media_hist_mm"] = np.where(df["Media_hist_mm"].notna(), df["Media_hist_mm"], df["Media_hist_mm_from_hist"])
        df.drop(columns=[c for c in df.columns if c.endswith("_from_hist")], inplace=True)
    except Exception as e:
        print("Aviso: no se pudo leer histórico para medias ->", e)

# 3) Fills en cascada para Pred_mm
# FILL A: usar Media_hist_mm cuando falte Pred_mm
mask = df["Pred_mm"].isna() & df["Media_hist_mm"].notna()
df.loc[mask, "Pred_mm"] = df.loc[mask, "Media_hist_mm"]

# FILL B: mediana por estado a partir de Pred_mm disponibles
state_med = df.groupby("STATE_ID")["Pred_mm"].median()
mask = df["Pred_mm"].isna() & df["STATE_ID"].notna()
df.loc[mask, "Pred_mm"] = df.loc[mask, "STATE_ID"].map(state_med)

# FILL C: mediana nacional de Pred_mm
nat_med = float(df["Pred_mm"].median(skipna=True))
df["Pred_mm"] = df["Pred_mm"].fillna(nat_med)

# 4) Recalcular Diff_pct (si hay Media_hist_mm)
df["Diff_pct"] = np.where(
    df["Media_hist_mm"].notna() & df["Pred_mm"].notna() & (df["Media_hist_mm"]!=0),
    100.0*(df["Pred_mm"] - df["Media_hist_mm"]) / df["Media_hist_mm"],
    np.nan
)

# 5) Escribir de vuelta en el GeoJSON
pred_ok = df["Pred_mm"].notna().sum()
for feat in fc["features"]:
    p = feat["properties"]
    code = str(p.get("ADM2_PCODE","")).strip().upper()
    row = df.loc[df["ADM2_PCODE"]==code].iloc[0]
    p["Pred_mm"] = float(row["Pred_mm"]) if pd.notna(row["Pred_mm"]) else None
    p["Media_hist_mm"] = float(row["Media_hist_mm"]) if pd.notna(row["Media_hist_mm"]) else None
    p["Diff_pct"] = float(row["Diff_pct"]) if pd.notna(row["Diff_pct"]) else None

json.dump(fc, open(GEO, "w", encoding="utf-8"))
print(f"✅ Pred_mm cubierto: {pred_ok}/{len(df)}  (esperado: 338)")

def strip_accents(s): 
    return "".join(c for c in unicodedata.normalize("NFKD", str(s)) if not unicodedata.combining(c))
def norm_name(s):
    s = strip_accents(str(s)).upper().strip()
    s = re.sub(r"\s+", " ", s)
    s = s.replace("MCPIO. ", "").replace("MCPIO ", "")
    s = s.replace("MUNICIPIO AUTONOMO ", "").replace("MUNICIPIO AUTÓNOMO ", "")
    return s

# 1) Cargar GeoJSON final
fc = json.load(open(GEO, "r", encoding="utf-8"))

def coerce_num(s: pd.Series) -> pd.Series:
    """Convierte a numérico soportando coma decimal."""
    s = s.astype(str)
    if (s.str.contains(r"^\s*-?\d+,\d+\s*$").head(500).mean() > 0.2):
        s = s.str.replace(",", ".", regex=False)
    return pd.to_numeric(s, errors="coerce")

def plot_feature_importances(model_path, feature_names=None, k=25, title="Importancia de variables"):
    if not Path(model_path).exists():
        print("⚠️ No existe:", model_path); return
    mdl = joblib.load(model_path)
    # Localiza el paso 'rf'
    if isinstance(mdl, Pipeline):
        rf = dict(mdl.named_steps).get("rf", None)
    else:
        rf = mdl if isinstance(mdl, RandomForestRegressor) else None
    if rf is None or not hasattr(rf, "feature_importances_"):
        print("⚠️ No encuentro RandomForest con feature_importances_."); return
    imps = rf.feature_importances_
    # nombres
    if feature_names is None:
        feature_names = getattr(mdl, "feature_names_in_", None)
    if feature_names is None or len(feature_names) != len(imps):
        feature_names = [f"f{i}" for i in range(len(imps))]
    # top-k
    order = np.argsort(imps)[::-1][:k]
    plt.figure(figsize=(8, max(3, k*0.28)))
    plt.barh(range(len(order)), imps[order][::-1], alpha=0.85)
    plt.yticks(range(len(order)), [feature_names[i] for i in order][::-1])
    plt.xlabel("Importancia (Gini)"); plt.title(title); plt.grid(True, axis="x", alpha=0.25)
    plt.tight_layout(); save_fig(Path(model_path).stem + "_importances"); plt.show()
